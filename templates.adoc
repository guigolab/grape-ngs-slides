= Template Scripts
icon:arrow-circle-o-right[role="crg",float="left"] the script [crg]#`templates/index/STAR`# is used

[source]
----
#!/bin/bash -eu
mkdir genomeDir
STAR --runThreadN ${cpus} \
     --runMode genomeGenerate \
     --genomeDir genomeDir \
     --genomeFastaFiles ${genome} \
     --sjdbGTFfile ${annotation} \
     --sjdbOverhang ${sjOverHang}
----
